Quick Start

  const { Aqua, Client } = require("discord.js");
  const client = new Client({ intents: ["Guilds", "GuildVoiceStates"] });
  const aqua = Aqua(client, [{ host: "127.0.0.1", password: "yourpass", port: 233, secure: false, name: "localhost" }], {
    defaultSearchPlatform: "ytsearch",
    restVersion: "v4"
  });
  client.aqua = aqua;

  client.once("ready", () => aqua.init(client.user.id));
  client.on("messageCreate", async (message) => {
    if (message.content.startsWith("!play")) {
      const query = message.content.slice(6);
      const player = aqua.createConnection({
     guildId: message.guild.id,
     voiceChannel: message.member.voice.channel.id,
     textChannel: message.channel.id
      });
      const resolve = await aqua.resolve({
     query,
     requester: message.member
      });
      if (resolve.loadType === 'track') {
     player.queue.add(resolve.tracks[0]);
     await message.channel.send(`Added **${resolve.tracks[0].info.title}** to the queue.`);
     if (!player.playing) player.play();
      }
    }
  });

  client.login("Yourtokenhere");

Aqua (Options)

Main client configuration and initialization options for the Aqua Lavalink client.

Constructor CLASS

Initialize a new Aqua client instance with custom options. 

const { Aqua } = require('aqualink');

const aqua = new Aqua({
nodes: [{
    host: 'localhost',
    port: 2333,
    password: 'youshallnotpass',
    secure: false
}],
defaultSearchPlatform: 'ytsearch,
shouldDeleteMessage: false,
leaveOnEnd: true,
restVersion: 'v4',
plugins: [],
autoResume: false,
infiniteReconnects: false,
failoverOptions: {
    enabled: true,
    maxRetries: 3,
    retryDelay: 1000,
    preservePosition: true,
    resumePlayback: true,
    cooldownTime: 5000,
    maxFailoverAttempts: 5
    }
});
Parameter Type Description Required
nodes Array<NodeOptions> Array of Lavalink node configurations Required
defaultSearchPlatform string Default platform for track searches Optional
shouldDeleteMessage boolean Delete messages after processing Optional
leaveOnEnd boolean Leave voice channel when queue ends Optional
restVersion string REST API version Optional
plugins Array Array of plugin instances Optional
autoResume boolean Automatically resume playback after reconnect Optional
infiniteReconnects boolean Enable infinite reconnect attempts Optional
failoverOptions object Failover and retry configuration Optional
init() METHOD

Initialize the Aqua client and connect to Lavalink nodes. 

        // Needs to be used before the client login
        client.on('ready', (client) => {
            // Initialize the client
            client.aqua.init(client.user.id);

            // Client is now ready to use
            console.log('Aqua client initialized!');
        });

Rest

REST API methods for interacting with Lavalink nodes directly. NOT MADE TO BE USED.

updatePlayer(options) METHOD

Update player state or settings via REST API.

await client.rest.updatePlayer({ guildId: "123456789012345678", paused: true, volume: 80 });
getPlayers() METHOD

Get all active players from the Lavalink node.

const players = await client.rest.getPlayers(); console.log('Players:', players);
destroyPlayer(guildId) METHOD

Destroy a player for a specific guild.

await client.rest.destroyPlayer("123456789012345678");
getTracks(identifier) METHOD

Get track(s) by identifier from Lavalink.

const tracks = await client.rest.getTracks("dQw4w9WgXcQ"); console.log('Tracks:', tracks);
decodeTrack(track) METHOD

Decode a base64 encoded track string to get track information.

const trackInfo = await client.rest.decodeTrack(encodedTrack); console.log(trackInfo.title, trackInfo.author);
decodeTracks(tracks) METHOD

Decode multiple base64 encoded tracks.

const trackInfos = await client.rest.decodeTracks([encodedTrack1, encodedTrack2]); console.log(trackInfos);
getStats() METHOD

Get current statistics from the Lavalink node.

const stats = await client.rest.getStats(); console.log('Players:', stats.players); console.log('Playing players:', stats.playingPlayers);
getInfo() METHOD

Get Lavalink node information.

const info = await client.rest.getInfo(); console.log('Node info:', info);
getRoutePlannerStatus() METHOD

Get route planner status from Lavalink.

const status = await client.rest.getRoutePlannerStatus(); console.log('Route planner status:', status);
getRoutePlannerAddress(address) METHOD

Unmark an address from the route planner.

await client.rest.getRoutePlannerAddress("1.2.3.4");

Player

Audio player controls and playback management methods.

createConnection(options) METHOD

Create a new player connection for a guild.

const player = client.aqua.createConnection({ guildId: "123456789012345678", textChannel: "123456789012345678", voiceChannel: "123456789012345678", deaf: true });
queue.add(track) METHOD

Add a track to the player's queue.

player.queue.add(track);
play() METHOD

Start playback of the current track in the queue.

player.play();
setVolume(volume) METHOD

Set the playback volume (0-1000).

player.setVolume(75);
pause(paused) METHOD

Pause or resume playback. Pass true to pause, false to resume.

player.pause(true); player.pause(false);
skip() METHOD

Skip to the next track in the queue.

player.skip();
shuffle() METHOD

Shuffle the queue.

player.shuffle();
setLoop(mode) METHOD

Set loop mode. Options: 'track', 'queue', 'none'.

player.setLoop('track'); player.setLoop('queue'); player.setLoop('none');
setAutoplay(enabled) METHOD

Enable or disable autoplay.

player.setAutoplay(true); player.setAutoplay(false);
destroy() METHOD

Destroy the player and disconnect.

player.destroy();
getLyrics([query], [useCurrentTrack], [skipTrackSource]) METHOD

Get lyrics for the current or specified track.

const lyrics = await player.getLyrics(); // Optional: query, useCurrentTrack = true, skipTrackSource = false
subscribeLiveLyrics() METHOD

Subscribe to live lyrics updates.

player.subscribeLiveLyrics();
unsubscribeLiveLyrics() METHOD

Unsubscribe from live lyrics updates.

player.unsubscribeLiveLyrics();
seek(position) METHOD

Seek to a specific position in the current track (milliseconds).

player.seek(60000);
stop() METHOD

Stop playback and clear the current track.

player.stop();
setTextChannel(channelId) METHOD

Set the text channel for player events.

player.setTextChannel("123456789012345678");
setVoiceChannel(channelId) METHOD

Set the voice channel for playback.

player.setVoiceChannel("123456789012345678");
disconnect() METHOD

Disconnect the player from the voice channel.

player.disconnect();
getQueue() METHOD

Get the current queue.

const queue = player.getQueue();
replay() METHOD

Replay the current track from the beginning.

player.replay();

Track

Track objects and methods for handling audio track information.

Track Object OBJECT

Represents an audio track with metadata and playback information.

const track = { encoded: 'QAAArQIAJVJpY2sgQXN0bGV5IC0gTmV2ZXIgR29ubmEgR2l2ZSBZb3UgVXA...', info: { identifier: 'dQw4w9WgXcQ', isSeekable: true, author: 'Rick Astley', length: 212000, isStream: false, position: 0, title: 'Never Gonna Give You Up', uri: 'https://www.youtube.com/watch?v=dQw4w9WgXcQ', sourceName: 'youtube' } };
Property Type Description
encoded string Base64 encoded track data
info.identifier string Unique track identifier
info.title string Track title
info.author string Track author/artist
info.length number Track duration in milliseconds
info.uri string Original track URI
info.sourceName string Source platform (youtube, soundcloud, etc.)
info.isSeekable boolean Whether the track supports seeking
info.isStream boolean Whether the track is a live stream

Events

Event listeners for handling player, track, lyrics, and node state changes.

trackStart EVENT

Fired when a track starts playing.

aqua.on('trackStart', (player, track) => { console.log('Now playing:', track.info.title); });
trackEnd EVENT

Fired when a track finishes playing.

aqua.on('trackEnd', (player, track, reason) => { console.log('Track ended:', reason); });
trackStuck EVENT

Fired when a track gets stuck during playback.

aqua.on('trackStuck', (player, track, thresholdMs) => { console.log('Track stuck for', thresholdMs, 'ms'); });
trackError EVENT

Fired when an error occurs during track playback.

aqua.on('trackError', (player, track, error) => { console.error('Track error:', error.message); });
socketClosed EVENT

Fired when the Lavalink socket is closed.

aqua.on('socketClosed', (player, code, reason) => { console.log('Socket closed:', code, reason); });
lyricsNotFound EVENT

Fired when lyrics are not found for a track.

aqua.on('lyricsNotFound', (track) => { console.log('Lyrics not found for:', track.info.title); });
lyricsFound EVENT

Fired when lyrics are found for a track.

aqua.on('lyricsFound', (track, lyrics) => { console.log('Lyrics found for:', track.info.title); });
lyricsLine EVENT

Fired for each line of live lyrics.

aqua.on('lyricsLine', (line) => { console.log('Lyrics line:', line); });
nodeConnect EVENT

Fired when a Lavalink node connects.

aqua.on('nodeConnect', (node) => { console.log('Node connected:', node.options.host); });
nodeDisconnect EVENT

Fired when a Lavalink node disconnects.

aqua.on('nodeDisconnect', (node, reason) => { console.log('Node disconnected:', node.options.host, reason); });
nodeError EVENT

Fired when a Lavalink node encounters an error.

aqua.on('nodeError', (node, error) => { console.error('Node error:', node.options.host, error.message); });
nodeCreate EVENT

Fired when a Lavalink node is created.

aqua.on('nodeCreate', (node) => { console.log('Node created:', node.options.host); });
playerCreate EVENT

Fired when a player is created.

aqua.on('playerCreate', (player) => { console.log('Player created for guild:', player.guildId); });
nodeDestroy EVENT

Fired when a Lavalink node is destroyed.

aqua.on('nodeDestroy', (node) => { console.log('Node destroyed:', node.options.host); });
playerDestroy EVENT

Fired when a player is destroyed.

aqua.on('playerDestroy', (player) => { console.log('Player destroyed for guild:', player.guildId); });
debug EVENT

Fired for debug messages.

aqua.on('debug', (message) => { console.debug('Debug:', message); });

Queue

Queue management methods for handling track playlists.

add(track) METHOD

Add a track to the queue.

const result = await client.rest.loadTracks('Never Gonna Give You Up'); if (result.tracks.length > 0) { player.queue.add(result.tracks[0]); console.log('Track added to queue'); }
remove(index) METHOD

Remove a track from the queue by index.

const removedTrack = player.queue.remove(1); console.log('Removed:', removedTrack.info.title);
clear() METHOD

Clear all tracks from the queue.

player.queue.clear(); console.log('Queue cleared');
shuffle() METHOD

Shuffle the tracks in the queue.

player.queue.shuffle(); console.log('Queue shuffled');
next() METHOD

Skip to the next track in the queue.

const nextTrack = player.queue.next(); if (nextTrack) { await player.play(nextTrack); } else { console.log('No more tracks in queue'); }
previous() METHOD

Go back to the previous track.

const previousTrack = player.queue.previous(); if (previousTrack) { await player.play(previousTrack); }
size PROPERTY

Get the current size of the queue.

console.log('Queue size:', player.queue.size);
isEmpty PROPERTY

Check if the queue is empty.

if (player.queue.isEmpty) { console.log('Queue is empty'); }
current PROPERTY

Get the currently playing track.

const currentTrack = player.current; if (currentTrack) { console.log('Now playing:', currentTrack.info.title); }
